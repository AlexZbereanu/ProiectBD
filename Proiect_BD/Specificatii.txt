	Jucatorii:
		 -id jucator
		 -nume(NOT NULL)
		 -prenume

		 -numar(constrans in intervalul 1 si 99)
		 -tara natala

	Echipele:
		-id echipa
		-nume club
		-prescurtare(ex: Inter Milano->"INT")(constrans la 3

		-id stadion
		-numele stadionului
		-capacitatea sa

	Meciurile:
		-id meci
		-data desfasurarii
		-numar spectatori
